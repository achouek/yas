{
  OBJECT-PROPERTIES
  {
    Date=23/09/13;
    Time=12:00:00;
    Version List=NAVW17.10;
  }
  PROPERTIES
  {
    TableNo=112;
    Permissions=TableData 112=rm,
                TableData 114=rm;
    OnRun=BEGIN
            CancelInvoice(Rec);
          END;

  }
  CODE
  {
    VAR
      CancelPostedInvoiceQst@1000 : TextConst 'ENU=The posted sales invoice will be canceled, and a sales credit memo will be created and posted, which reverses the posted sales invoice.\ \Do you want to continue?;FRA=La facture vente enregistr‚e sera annul‚e et un avoir vente sera cr‚‚ et enregistr‚, lequel annule la facture vente enregistr‚e.\ \Souhaitez-vous continuerÿ?';
      OpenPostedCreditMemoQst@1001 : TextConst 'ENU=A credit memo was successfully created. Do you want to open the posted credit memo?;FRA=Un avoir a ‚t‚ correctement cr‚‚. Voulez-vous ouvrir l''avoir enregistr‚ÿ?';

    PROCEDURE CancelInvoice@1(VAR SalesInvoiceHeader@1002 : Record 112) : Boolean;
    VAR
      SalesCrMemoHeader@1004 : Record 114;
      CorrectPostedSalesInvoice@1003 : Codeunit 1303;
    BEGIN
      CorrectPostedSalesInvoice.TestCorrectInvoiceIsAllowed(SalesInvoiceHeader,TRUE);
      IF CONFIRM(CancelPostedInvoiceQst) THEN
        IF CorrectPostedSalesInvoice.CancelPostedInvoice(SalesInvoiceHeader) THEN
          IF CONFIRM(OpenPostedCreditMemoQst) THEN BEGIN
            SalesCrMemoHeader.GET(SalesInvoiceHeader."Canceled By");
            PAGE.RUN(PAGE::"Mini Posted Sales Credit Memo",SalesCrMemoHeader);
            EXIT(TRUE);
          END;

      EXIT(FALSE);
    END;

    BEGIN
    END.
  }
}
